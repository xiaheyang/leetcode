    string getPermutation(int n, int k) {
        int allnum=1;
        string result="";
        for(int i=0;i<n;++i)
        {
            allnum=allnum*(i+1);
            result.push_back('0'+i+1);
        }
        int pos=n;
        k=k-1;//1.¹éÁã
        while(pos>0)
        {
            allnum=allnum/pos;
            int changepos=k/allnum;
            //²âÊÔÏÂ¸´ÔÓ¶È
            // swap(result[n-pos],result[n-pos+changepos]);//1.Î»ÖÃ´íÎó
            // sort(result.begin()+n-pos+1,result.end());
            result.insert(result.begin()+n-pos,result[n-pos+changepos]);
            result.erase(result.begin()+n-pos+changepos+1);
            k=k%allnum;
            --pos;
        }
        return result;
    }
